import { Router } from "express";
import {
  validateCreateBaseResource,
  validateGetBaseResources,
  validateGetBaseResource,
  validateDeleteBaseResource,
  validateUpdateBaseResource,
} from "../validators/BaseResourcesValidator";

import {
  createBaseResource,
  getBaseResources,
  getBaseResource,
  updateBaseResource,
  deleteBaseResource,
} from "../controllers/BaseResourcesController";

const router = Router();

// @desc    Create a baseResource
// @route   POST /api/v1/baseResources
// @access  Private
router.post("/", validateCreateBaseResource, createBaseResource);

// @desc    Get baseResources data
// @route   GET /api/v1/baseResources
// @access  Private
router.get("/", validateGetBaseResources, getBaseResources);

// @desc    Get a baseResource data
// @route   GET /api/v1/baseResources/:id
// @access  Private
router.get("/:id", validateGetBaseResource, getBaseResource);

// @desc    Update a baseResource data
// @route   PUT /api/v1/baseResources/:id
// @access  Private
router.put("/:id", validateUpdateBaseResource, updateBaseResource);

// @desc    Delete a baseResource data
// @route   PUT /api/v1/baseResources/:id
// @access  Private
router.delete("/:id", validateDeleteBaseResource, deleteBaseResource);

export { router };

import { Schema, model } from "mongoose";
import { BaseResource } from "../interfaces/interfaces";

const BaseResourceSchema = new Schema <BaseResource>(  
  {
    school_id: {
      type: Schema.Types.ObjectId,
      ref: "School",
      required: [true, "must provide a school id for the baseResource"],
    },
    name: {
      type: String,
      required: [true, "must provide name for the baseResource"],
    },
  },
  {
    timestamps: true,
    versionKey: false,
  }
);

const BaseResourceModel = model <BaseResource>( "BaseResource", BaseResourceSchema);

export default BaseResourceModel;

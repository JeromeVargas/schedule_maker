import { NextFunction, Request, Response } from "express";
import { check } from "express-validator";
import validateResult from "../../lib/helpers/validateHelper";
import { isValidId } from "../../lib/utilities/utils";

// @fields: body {school_id:[string], name:[string]}
const validateCreateTemplate = [
  check("school_id")
    .exists()
    .withMessage("Please add the school id")
    .bail()
    .notEmpty()
    .withMessage("The school field is empty")
    .bail()
    .custom((value) => {
      const validId = isValidId(value);
      if (validId === false) {
        return false;
      } else if (validId === true) {
        return true;
      }
    })
    .withMessage(`The school id is not valid`),
  check("name")
    .exists()
    .withMessage("Please add a template name")
    .bail()
    .notEmpty()
    .withMessage("The template name field is empty")
    .bail()
    .isString()
    .withMessage("The template name is not valid")
    .isLength({ min: 1, max: 100 })
    .withMessage("The template name must not exceed 100 characters")
    .escape()
    .trim(),
  (req: Request, res: Response, next: NextFunction) => {
    validateResult(req, res, next);
  },
];

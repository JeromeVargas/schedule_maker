import { Router } from "express";
import {
  validateCreateTemplate,
  validateGetTemplates,
  validateGetTemplate,
  validateUpdateTemplate,
  validateDeleteTemplate,
} from "./templatesValidator";

import {
  createTemplate,
  getTemplates,
  getTemplate,
  updateTemplate,
  deleteTemplate,
} from "./templatesController";

const router = Router();

// @desc    Create a template
// @route   POST /api/v1/templates
// @access  Private
router.post("/", validateCreateTemplate, createTemplate);

// @desc    Get templates data
// @route   GET /api/v1/templates
// @access  Private
router.get("/", validateGetTemplates, getTemplates);

// @desc    Get a template data
// @route   GET /api/v1/templates/:id
// @access  Private
router.get("/:id", validateGetTemplate, getTemplate);

// @desc    Update a template data
// @route   PUT /api/v1/templates/:id
// @access  Private
router.put("/:id", validateUpdateTemplate, updateTemplate);

// @desc    Delete a template
// @route   PUT /api/v1/templates/:id
// @access  Private
router.delete("/:id", validateDeleteTemplate, deleteTemplate);

export { router };
